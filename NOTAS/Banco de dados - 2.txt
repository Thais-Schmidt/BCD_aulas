SELECT CONCAT(CLI.NOME, ' ', CLI.SOBRENOME) AS NOME_COMPLETO,
CLI.SEXO, CLI.PESO,
GROUP_CONCAT(TEL.NUMERO) AS TELEFONES
FROM TBL_CLIENTES AS CLI
INNER JOIN TBL_TELEFONES AS TEL
ON CLI.ID=TEL.ID_CLIENTE

SELECT CLI.ID,
